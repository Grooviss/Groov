#Apskaiciuoja kortu verte
def hand_value (hand):
    value = 0
    aces = 0
    for card in hand:
        if card == 'A':
            aces += 1
            value += 11
        elif card in ['K', 'Q', 'J']:
            value += 10
        else:
            value += card
    while value > 21 and aces > 0:
        value -= 10
        aces -= 1
    return value
